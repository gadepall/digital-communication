\noindent \textbf{11.16.3.14.} Given $\pr{A}=\frac{3}{5}$ and $\pr{B}=\frac{1}{5}$. Find $\pr{A+B}$ if $A$ and $B$ are mutually exclusive events.\\

\noindent\textbf{solution:-} Given:
\begin{align}
\pr{A}&=\frac{3}{5} \\
\pr{B}&=\frac{1}{5}\\
\pr{AB}&=0 
\end{align}
\begin{align}
  \pr{A+B} &= \pr{A} + \pr{B} - \pr{AB}\\ 
  &= \frac{3}{5} + \frac{1}{5} - 0 \\
  &= \frac{4}{5}
\end{align}

   $$ \therefore \pr{A+B}=\frac{4}{5} $$
