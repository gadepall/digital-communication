\textbf{Question: } Two balls are drawn at random with replacement from a box containing 10 black balls and 8 red balls. Find the probability that

\begin{enumerate}[label=(\roman{enumi})]
\item both balls are red
\item first ball is black and second is red
\item one of them is black and other is red
\end{enumerate}

\textbf{Solution: }
Since balls are drawn with replacement, Each draw is independent of other draw. Probability of getting red ball $p$ in any draw is
\begin{align}
\label{eq:1}
p=\frac{8}{18}=\frac{4}{9}
\end{align}
$X$, $X_1$ and $X_2$ are the random variables for binomial distribution and two bernoulli trials respectively.
\begin{align}
	X&=X_1+X_2\\
	p_{X_i}(k) &= 
	\begin{cases}
		1-p, & k=0 \\
		p  , & k=1 \\
		0  , & \text{otherwise}\\
	\end{cases}\\
	i&\in{\{1 ,2\}}
\end{align}
\begin{align}
	\label{eq:2}
	p_{X}(r)=\comb{n}{r}p^{r}(1-p)^{n-r}
\end{align}
Here, $n=2$ and $p=\frac{4}{9}$
\begin{align}
	X&\in{\{0, 1 ,2\}} \\
	X &\sim Bin\brak{2,\frac{4}{9}}
\end{align}
From \eqref{eq:1} and \eqref{eq:2},
\begin{align}
	\therefore~ p_{X}(r)&=\comb{2}{r} \brak{\frac{4}{9}}^{r} \brak{\frac{5}{9}}^{2-r}
\end{align}
\begin{align}
	p_X(k) = 
	\begin{cases}
		\frac{25}{81}, & k=0 \\
		\frac{40}{81}, & k=1 \\
		\frac{16}{81}, & k=2 \\
		0, & \text{otherwise}\\
	\end{cases}
\end{align}
\begin{enumerate}[label=(\roman{enumi})]
\item
	Since both balls are red, the required probability is $p_{X}(2)$
	\begin{align}		
	\therefore~ p_{X}(2)&=\frac{16}{81}
	\end{align}
	
\item
	Let $E$ be the event that first ball is black and other is red.
	The required probability \pr{E} is
	\begin{align}
		\pr{E}&=p_{X_1}(0)\times p_{X_2}(1)\\
			&=\frac{1}{2} \times p_{X}(1)\\
			&=\frac{1}{2} \times \comb{2}{1} \brak{\frac{4}{9}}^{1} \brak{\frac{5}{9}}^{1}\\
		&=\frac{1}{2} \times 2 \times \frac{4}{9} \times \frac{5}{9}\\
			&=\frac{20}{81}\\
		\therefore~ \pr{E}&=\frac{20}{81}
	\end{align}
\item
	Since only one of the two balls has to be red, The required probability is $p_{X}(1)$ 
	\begin{align}
	\therefore~ p_{X}(1)&=\frac{40}{81}
	\end{align}
	
\end{enumerate}
